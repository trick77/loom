# Projects Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Projects as a first-class chat organization feature where each chat can belong to one project or to no project.

**Architecture:** Keep the existing nullable `threads.project_id` as the membership source of truth and extend the existing thread update API to move chats into and out of projects. Add focused React modules under `ui/src/projects/` for project pages, picker UI, and project membership helpers; `ui/src/chat/ChatShell.tsx` remains a route/data coordinator and must not become a project god component.

**Tech Stack:** Go 1.25, stdlib `net/http`, SQLite through `ncruces/go-sqlite3`, React, TypeScript, Vite, Vitest, Testing Library, Tailwind with Lume theme tokens.

## Component Boundaries

Do not create god classes or god components. In this codebase that means:

- `ChatShell` coordinates route state, shared loaded data, and high-level callbacks only.
- Project page JSX lives in `ui/src/projects/ProjectsPage.tsx` and `ui/src/projects/ProjectDetailPage.tsx`.
- Project modal JSX lives in dedicated modal files.
- Repeated membership-list transformations live in `ui/src/projects/projectMembership.ts`.
- If a new project file grows beyond one clear responsibility during implementation, split it before committing.
- Do not hide broad project behavior in one giant hook with unrelated state; hooks/helpers must have narrow names and narrow jobs.

```go
func (s *Store) UpdateThread(ctx context.Context, userID, threadID string, in UpdateThreadInput) (Thread, bool, error) {
	thread, ok, err := s.GetThread(ctx, userID, threadID)
	if err != nil || !ok {
		return Thread{}, ok, err
	}

	title := thread.Title
	if in.Title != nil {
		title = NormalizeThreadTitle(*in.Title)
		if title == "" {
			return Thread{}, false, errors.New("thread title is required")
		}
	}

	var projectID any
	if thread.ProjectID != nil {
		projectID = *thread.ProjectID
	}
	if in.ProjectID.Set {
		projectID = nil
		if in.ProjectID.Value != nil {
			if ok, err := s.projectExists(ctx, userID, *in.ProjectID.Value); err != nil {
				return Thread{}, false, err
			} else if !ok {
				return Thread{}, false, errors.New("project not found")
			}
			projectID = *in.ProjectID.Value
		}
	}

	_, err = s.db.ExecContext(ctx, `
UPDATE threads
SET title = ?, project_id = ?, updated_at = datetime('now')
WHERE user_id = ? AND id = ?`,
		title, projectID, userID, threadID,
	)
	if err != nil {
		return Thread{}, false, fmt.Errorf("update thread: %w", err)
	}
	return s.GetThread(ctx, userID, threadID)
}
```

```go
func (r *updateThreadRequest) UnmarshalJSON(data []byte) error {
	var raw struct {
		Title     *string          `json:"title"`
		ProjectID *json.RawMessage `json:"projectId"`
	}
	if err := json.Unmarshal(data, &raw); err != nil {
		return err
	}
	r.Title = raw.Title
	if raw.ProjectID == nil {
		return nil
	}
	r.ProjectIDProvided = true
	if string(*raw.ProjectID) == "null" {
		r.ProjectID = nil
		return nil
	}
	var projectID string
	if err := json.Unmarshal(*raw.ProjectID, &projectID); err != nil {
		return err
	}
	r.ProjectID = &projectID
	return nil
}

func (r updateThreadRequest) chatInput() chat.UpdateThreadInput {
	return chat.UpdateThreadInput{
		Title: r.Title,
		ProjectID: chat.ProjectIDUpdate{
			Set:   r.ProjectIDProvided,
```

```ts
export type RouteState =
  | { view: "new" }
  | { view: "chats" }
  | { view: "chat"; threadID: string }
  | { view: "projects" }
  | { view: "project"; projectID: string };

export function routeFromPath(path: string): RouteState {
  if (path.startsWith("/chat/")) {
    const threadID = decodeURIComponent(path.slice("/chat/".length));
    if (threadID !== "") return { view: "chat", threadID };
  }
  if (path === "/chats") return { view: "chats" };
  if (path === "/projects") return { view: "projects" };
  if (path.startsWith("/projects/")) {
    const projectID = decodeURIComponent(path.slice("/projects/".length));
    if (projectID !== "") return { view: "project", projectID };
  }
  return { view: "new" };
}

export function routeFromLocation(): RouteState {
  return routeFromPath(window.location.pathname);
}

export function pathForRoute(route: RouteState): string {
  if (route.view === "new") return "/new";
  if (route.view === "chats") return "/chats";
  if (route.view === "projects") return "/projects";
  if (route.view === "project") return `/projects/${encodeURIComponent(route.projectID)}`;
  return `/chat/${encodeURIComponent(route.threadID)}`;
}

export function navigate(route: RouteState) {
  const path = pathForRoute(route);
  if (window.location.pathname !== path) {
    window.history.pushState({}, "", path);
  }
}
```
